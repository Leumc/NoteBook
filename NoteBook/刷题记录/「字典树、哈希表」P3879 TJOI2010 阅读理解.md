# 问题分析：

既然涉及到了记录出现频率的问题，那么最先想到的方法自然是使用哈希表，以`string`为键，以`set<int>`为值，即可轻松解决问题。`set`具有升序和无重复的特性，这能够节省很大的代码量。这里直接贴出代码：

```cpp
#include <bits/stdc++.h>
using namespace std;
map<string,set<int>> mp;
int n,m;
string s;
int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>m;
        for(int j=1;j<=m;j++){
            cin>>s;
            if(mp.find(s)==mp.end()){
                set<int> v={i};
                mp.insert(make_pair(s,v));
            }
            else{
                mp[s].insert(i);
            }
        }
    }
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>s;
        if(mp.find(s)!=mp.end()){
            for(int n:mp[s]){
                cout<<n<<" ";
            }
        }
        cout<<endl;
    }
}
```

# 字典树：

正如它的名字所示，它的作用是查询，实现方式是树。它是一个多叉树，每一个节点上都有着一个字符，顺着一定的路径把节点上的字符拼接起来，便可以组成一个字符串，如下图所示：

![字典树|697](https://cdn.luogu.com.cn/upload/pic/15735.png)

面向对象实现字典树：

```cpp
class Trie{
private:
    struct TrieNode
    {
        unordered_map<char,TrieNode*> children;//储存字符对应的子节点
        set<int> ids;//储存当前字符串出现行的行号
        bool is_end;//当前位置是否是某个字符串的结尾
        TrieNode():is_end(false){}//初始化is_end为false
    };
    TrieNode* root;
public:
    void insert(string s,int i){
        TrieNode* node=root;
        for(char c:s){
            //存一个引用方便调用
            unordered_map<char,TrieNode*> &cd=node->children;
            //如果没有找到下一个字符对应的子节点，就创建一个
            if(cd.find(c)==cd.end()){
                cd.insert(make_pair(c,new TrieNode()));
            }
            //进入下一个字符对应节点
            node=cd[c];
        }
        node->ids.insert(i);
        node->is_end=true;
    }
    //查询一个字符串是否存在，如果存在的化就通过st引用变量将ids带出
    bool find(string s,set<int> &st){
        TrieNode* node=root;
        for(char c:s){
            unordered_map<char,TrieNode*> &cd=node->children;
            if(cd.find(c)==cd.end()){
                return false;
            }
            node=cd[c];
        }
        if(node->is_end){
            st=node->ids;
            return true;
        }
    }
};
//...
```

竞赛常用的静态版本：

```cpp
#define MAX 5000005
int ts[MAX][26];//ts[u][c]表示编号u字符的下一个字符c的编号
int tot;//计数器，用于分配编号
set<int> ids[MAX];//记录出现行的行号
bool is_end[MAX];//记录编号位置是否是某个字符串的末尾
void insert(string s,int i){
    int u=0;//编号为0表示当前在根节点
    for(char c:s){
        //如果下一个字符没有编号，就为其分配一个
        if(!ts[u][c]){
            ts[u][c]=++tot;
        }
        u=ts[u][c];
    }
    is_end[u]=true;
    ids[u].insert(i);
}
//由于静态实现有统一的编号，所以可以直接通过编号来访问，如果返回-1则代表未找到
int find(string s){
    int u=0;
    for(char c:s){
        if(!ts[u][c])return -1;
        u=ts[u][c];
    }
    return is_end[u]?u:-1;
}
```

```cpp
#include <iostream>
#include <set>
#include <queue>
#include <algorithm>
using namespace std;
vector<pair<long long,vector<int>>> ns;//这样嵌套就不报错，也不知道为啥
const int P=31;const int M=1e9+7;
long long hash_key(string s){//计算哈希值
	long long key=0;
	for(char c:s){
		key=(key*P+(c-'a'+1))%M;
	}
	return key;
}
int main(){
	int n;cin>>n;
	for(int i=1;i<=n;i++){
		int l;cin>>l;
		for(int j=1;j<=l;j++){
			string s;cin>>s;
			long long key=hash_key(s);
			auto it=ns.begin();
			bool found=false;
			while(it!=ns.end()){
				cout<<(*it).first<<" ";
				if((*it).first==key){
					if(find((*it).second.begin(),(*it).second.end(),i)==(*it).second.end())(*it).second.push_back(i);//逆天嵌套
					found=true;
					break;
				}
				else{
					it++;
				}
			}
			if(!found){
				vector<int> v={i};
				ns.push_back(make_pair(key,v));
			}
		}
	}
	for(auto p:ns){
		sort(p.second.begin(),p.second.end());
	}
	int m;cin>>m;
	for(int i=1;i<=m;i++){
		string s;cin>>s;
		long long key=hash_key(s);
		for(auto p:ns){
			if(p.first==key){
				for(int n:p.second){
					cout<<n<<" ";
				}
			}
		}
		cout<<endl;
	}
}
```
