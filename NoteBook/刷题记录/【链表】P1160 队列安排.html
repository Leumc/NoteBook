问题分析：

总之，这道题就是在于找到这么一个结构，它能够十分便利地进行插入和删除操作，毫无疑问就是用链表来做。

#include <bits/stdc++.h>
using namespace std;
int n, m;
vector<int> ns;
int main() {
	cin >> n;
	ns.push_back(1);
	for (int i = 2; i <= n; i++) {
		int k, p; cin >> k >> p;
		auto it = ns.begin();
		while (it != ns.end()) {
			if (*it == k) {//寻找序号k所在位置
				ns.insert(it + p, i);
				break;
			}
			else it++;
		}
	}
	cin >> m;
	for (int i = 1; i <= m; i++) {
		int x; cin >> x;
		auto it = ns.begin();
		while (it != ns.end()) {
			if (*it == x) {//寻找序号x所在位置
				ns.erase(it);
				break;
			}
			else it++;
		}
	}
	for (int nn : ns) {
		cout << nn << " ";
	}
}

当然，这样写肯定是TLE的，这两个while循环找k的操作的时间复杂度都是$O(n^2)$级别的。于是呢，没啥办法看了看题解。嗯，确实是用链表实现的。

指针结构体实现链表：

我最早在LeetCode接触过的就是这么实现的，定义一个结构体Node，它有两个Node指针成员，分别指向该节点左右两端的节点（记得初始化为nullptr）。同时，它还应该有一个成员变量用来储存该节点的编号。 我们最多需要1e5个这样的节点，所以创建一个节点数组，用索引值代表它的编号，即：

1
2
3
4
5
struct Node{
	Node* left = nullptr;
	Node* right = nullptr;
	int num;
} Nodes[100005];

随后，读入N，我们就知道了实际的节点数目，对这前N个节点的编号进行设置：

1
for(int i = 1; i <= n; i++)Nodes[i]=i;

然后，我们就需要考虑一下插入和删除操作应该怎么实现了。 首先是插入，我们需要做两件事： 1. 令插入节点的左右指针指向插入位置左右两侧的两个节点； 2. 令左右两侧节点互相指向的指针指向新插入的节点。 另外，如果插入位置在首尾（即左右指针中有nullptr），那么只需要建立首/尾节点与新节点的互相指向关系即可。

非首尾插入操作图示（向节点k的右侧插入，节点n序号未知）如下：

Nodes[i].right->left = &Nodes[i];

然后再来考虑一下删除操作的实现。并不一定要真的把这个节点删掉，只要把它两边的链接断开即可，图示如下（删除节点i，节点k与节点n的序号未知）：

代码表示：

1
2
Nodes[Nodes[i].left] = Nodes[i].right;
Nodes[Nodes[i].right] = Nodes[i].left;

除此之外，由于题目中说了，可能会出现删除已经删除过的节点的情况，所以需要进行标记。而已经删除的节点一定不会出现在最终输出结果中，换言之，我们并不需要获取它的编号，所以可以直接将该节点的num成员设为-1来表示已删除。下一次删除遇到num = -1的节点时，直接continue即可。

经由上述操作，插入与删除都可以在$O(1)$的时间复杂度内完成。

然后在输出阶段，就会遇到这么一个问题，那就是不知道究竟应该从哪个节点开始输出（即不知道哪一个是首节点）。所以我们需要进行一点小小的修改，增加一个变量来追踪首节点位置。由于Nodes[0]没有用到过，所以我们可以也直接以Nodes[0].right指向的位置作为首节点。 随后在每一次首节点变更的时候更新Nodes[0].right或者追踪变量，能够引起首节点变化的情况有如下两种： 1. 向首节点左侧插入节点，首节点变为新插入节点； 2. 删除首节点，首节点变为首节点的后一个节点。

#include <bits/stdc++.h>
using namespace std;
int n, m;
struct node {
	node* left = nullptr;
	node* right = nullptr;
	int num;
}ns[100005];
int main() {
	cin >> n;
	for (int i = 1; i <= n; i++) {
		ns[i].num = i;
	}
	int start = 1;
	ns[0].right = &ns[start];
	for (int i = 2; i <= n; i++) {
		int k, p; cin >> k >> p;
		if (p == 1) {
			if (ns[k].right == nullptr) {
				ns[k].right = &ns[i];
				ns[i].left = &ns[k];
			}
			else {
				ns[i].left = &ns[k];
				ns[i].right = ns[k].right;
				ns[k].right->left = &ns[i];
				ns[k].right = &ns[i];
			}
		}
		else {
			if (ns[k].left == nullptr) {
				ns[k].left = &ns[i];
				ns[i].right = &ns[k];
				//首节点更新判断（情况1）
				if (k == start) {
					ns[0].right = &ns[i];
					start = i;
				}
			}
			else {
				ns[i].right = &ns[k];
				ns[i].left = ns[k].left;
				ns[k].left->right = &ns[i];
				ns[k].left = &ns[i];
			}
		}
	}
	cin >> m;
	for (int i = 1; i <= m; i++) {
		int x; cin >> x;
		if (ns[x].num == -1)continue;
		//首节点更新判断（情况2）
		if (x == start)ns[0].right = ns[x].right;
		if (ns[x].left == nullptr) {
			ns[x].right->left = nullptr;
		}
		else if (ns[x].right == nullptr) {
			ns[x].left->right = nullptr;
		}
		else {
			ns[x].left->right = ns[x].right;
			ns[x].right->left = ns[x].left;
		}
		ns[x].num = -1;
	}
	node* nd = &ns[0];
	while (true) {
		if (nd->right == nullptr)break;
		else {
			cout << nd->right->num << " ";
			nd = nd->right;
		}
	}
}

#include <bits/stdc++.h>
using namespace std;
int n, m;
struct node {
	int left;
	int right;
	bool deleted;
}ns[100005];
int main() {
	cin >> n;
	int start = 1;
	for (int i = 2; i <= n; i++) {
		int k, p; cin >> k >> p;
		if (p == 0) {
			if (ns[k].left == 0) {
				ns[k].left = i;
				ns[i].right = k;
				if (k == start) {
					start = i;
				}
			}
			else {
				ns[i].right = k;
				ns[i].left = ns[k].left;
				ns[k].left = i;
				ns[ns[i].left].right = i;
			}
		}
		else {
			if (ns[k].right == 0) {
				ns[k].right = i;
				ns[i].left = k;
			}
			else {
				ns[i].left = k;
				ns[i].right = ns[k].right;
				ns[k].right = i;
				ns[ns[i].right].left = i;
			}
		}
	}
	cin >> m;
	for (int i = 1; i <= m; i++) {
		int x; cin >> x;
		if (ns[x].deleted)continue;
		if (x == start)start = ns[x].right;
		if (ns[x].left == 0) {
			ns[ns[x].right].left = 0;
		}
		else if (ns[x].right == 0) {
			ns[ns[x].left].right = 0;
		}
		else {
			ns[ns[x].left].right = ns[x].right;
			ns[ns[x].right].left = ns[x].left;
		}
		ns[x].deleted = true;
	}
	cout << start << " ";
	while (true) {
		if (ns[start].right==0)break;
		cout << ns[start].right << " ";
		start = ns[start].right;
	}
}

静态数组实现：

#include <bits/stdc++.h>
using namespace std;
int n, m;
int l[100005], r[100005];
bool deleted[100005];
int main() {
	int start = 1;
	cin >> n;
	for (int i = 2; i <= n; i++) {
		int k, p; cin >> k >> p;
		if (p == 0) {
			if (l[k] == 0) {
				l[k] = i;
				r[i] = k;
				if (k == start) {
					start = i;
				}
			}
			else {
				r[i] = k;
				l[i] = l[k];
				r[l[k]] = i;
				l[k] = i;
			}
		}
		else {
			if (r[k] == 0) {
				r[k] = i;
				l[i] = k;
			}
			else {
				l[i] = k;
				r[i] = r[k];
				l[r[k]] = i;
				r[k] = i;
			}
		}
	}
	cin >> m;
	for (int i = 1; i <= m; i++) {
		int x; cin >> x;
		if (deleted[x])continue;
		if (x == start)start = r[x];
		if (r[x] == 0) {
			r[l[x]] = 0;
		}
		else if (l[x] == 0) {
			l[r[x]] = 0;
		}
		else {
			r[l[x]] = r[x];
			l[r[x]] = l[x];
		}
		deleted[x] = true;
	}
	cout << start << " ";
	while (true) {
		if (r[start] == 0) break;
		cout << r[start] << " ";
		start = r[start];
	}
}
